/***************************************************************
 üü¢ 6Ô∏è‚É£ WIN OR LOSE (Max number ‚Üí even/odd)
***************************************************************/
import java.util.*;
class WorkAtTech {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int max = Integer.MIN_VALUE;

        for(int i = 0; i < n; i++){
            max = Math.max(max, sc.nextInt());
        }
        System.out.println(max % 2 == 0 ? "WON" : "LOST");
    }
}
